	CIS_droids_divisions = {
	
		icon = generic_intelligence_operation

		allowed = {
			original_tag = CIS
		}

		available = {
			custom_trigger_tooltip = {
				tooltip = "[?ROOT.GetName] amount of credits atleast 5"
				check_variable = {
					var = creditsamount
					value = 5
					compare = greater_than_or_equals
				}
			}
		}

		cost = 0
		days_remove = 0
		
		visible = {
			always = yes
		}

		complete_effect = {
			custom_effect_tooltip = droids_divisions_tt
			subtract_from_variable = { creditsamount = 5 }
			if = {
				limit = {
					NOT = { has_country_flag = OOB_LOADED_B1 }
				}
				hidden_effect = {
					load_oob = "droid_troops"
					set_country_flag = OOB_LOADED_B1
					set_division_template_lock = {
						division_template = "Battle Droid Batallion"
						is_locked = yes
					}
				}
			}
			else = {
				capital_scope = {
					create_unit = {
						division = "name = \"Battle Droid Batallion\" start_experience_factor = 0 start_equipment_factor = 1"
						owner = ROOT
					}
				}
			}
		}

		ai_will_do = {
			factor = 1
			modifier = { num_divisions > 100 factor = 0 }
		}
	}
