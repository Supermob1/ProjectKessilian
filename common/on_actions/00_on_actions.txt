# yearly on_action by Yard1

on_actions = {
	on_startup = {
		effect = {
			set_global_flag = Republic_Era
			set_variable = { UTI.HODday = 1 }
			set_variable = { UTI.HODmonth = 1 }
			set_variable = { UTI.HODyear = -24 }
			set_variable = { UTI.weekday = 1 }
			set_variable = { UTI.bby = 0 }
		}
	}
	# this is executed on the 1st of every year FOR ONE RANDOM COUNTRY - use every_country inside
	on_daily = {
		effect = {
			UTI = {
				add_to_variable = { UTI.HODday = 1 }
				add_to_variable = { UTI.weekday = 1 }
				if = {
					limit = {
						AND = {
							check_variable = { UTI.HODday = 19 }
							check_variable = { UTI.HODmonth = 11 }
						}
					}
					set_variable = { UTI.HODday = 1 }
					set_variable = { UTI.HODmonth = 1 }
					add_to_variable { UTI.HODyear = 1 }
				}
				if = {
					limit = {
						check_variable = { UTI.HODday = 36 }
					}
					add_to_variable = { UTI.HODmonth = 1 }
					set_variable = { UTI.HODday = 1 }
				}
				if = {
					limit = {
						check_variable = { UTI.weekday = 6 }
					}
					set_variable = { UTI.weekday = 1 }
				}
				if = {
					limit = {
						check_variable = { UTI.HODyear = 0 }
					}
					set_variable = { UTI.bby = 1 }
				}
			}
			if = {
				# don't touch!
				limit = { 
					NOT = { 
						check_variable = { global.last_year = global.year }
						has_global_flag = on_yearly_done
					}
				}
				# this is done only ONCE per year, even though on_daily fires for every country
				# will also be done on startup

				# your code here

				
				# don't touch!
				set_variable = { global.last_year = global.year }
				set_global_flag = { flag = on_yearly_done value = 1 days = 1 }
			}
		}
	}
}
